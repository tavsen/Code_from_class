#include <iostream>

using namespace std;

int main()
{
unsigned  long long int a=1, b=2, u, a2, b2, i=0;   // i - счетчик нужных дробей

    for (int k=1; k<1000; k++) {
        a = a+b*2; //прибавляем 2
        u=b;        //переворачиваем
        b=a;
        a=u;
        if (a > 1000000000000000 or b > 1000000000000000) {
                a/=10;
                b/=10;
        }
        a2=a+b;   //прибавляем 1
        b2=b;
        cout << a2 << " / " << b2;
        for (int h=1; h>0; h++) { //проверяем больше ли цифр в числителе
            a2/=10;  //числитель
            b2/=10;  //знаменатель

            if (b2==0 and a2!=0) {
                i++;
                cout << "    " << i;
            }
            if (b2==0) break;
        }
        cout << endl;
    }
    return 0;
}
